361/*
362 * Auxiliary function to add an outgoing edge to a vertex (this),
363 * with a given destination vertex (d) and edge weight (w).
364 */
365template <class T>
366Edge<T>* Vertex<T>::addEdge(Vertex<T> *d, double capacity, int domain) {
367 auto newEdge = new Edge<T>(this, d, capacity, domain);
368 adj.push_back(newEdge);
369 d->incoming.push_back(newEdge);
370 return newEdge;
371}
372
373/*
374 * Auxiliary function to remove an outgoing edge (with a given destination (d))
375 * from a vertex (this).
376 * Returns true if successful, and false if such edge does not exist.
377 */
378
379template<class T>
380bool Vertex<T>::removeEdge(const T &id) {
381 bool removedEdge = false;
382 auto it = adj.begin();
383 while (it != adj.end()) {
384 Edge<T> *edge = *it;
385 Vertex *dest = edge->getDest();
386 if (dest->getInfo() == id) {
387 it = adj.erase(it);
388 deleteEdge(edge);
389 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
390 }
391 else {
392 it++;
393 }
394 }
395 return removedEdge;
396}
397
398/*
399 * Auxiliary function to remove an outgoing edge of a vertex.
400 */
401template<class T>
402void Vertex<T>::removeOutgoingEdges() {
403 auto it = adj.begin();
404 while (it != adj.end()) {
405 Edge<T> *edge = *it;
406 it = adj.erase(it);
407 deleteEdge(edge);
408 }
409}

451template<class T>
452void Vertex<T>::deleteEdge(Edge<T> *edge) {
453 Vertex<T> *dest = edge->getDest();
454 // Remove only the matching edge pointer from the incoming list.
455 auto it = dest->incoming.begin();
456 while (it != dest->incoming.end()) {
457 if (*it == edge) {
458 it = dest->incoming.erase(it);
459 }
460 else {
461 it++;
462 }
463 }
464 delete edge;
465}

535/*
536 * Adds a vertex with a given content or info (in) to a graph (this).
537 * Returns true if successful, and false if a vertex with that content already exists.
538 */
539
540template<class T>
541bool Graph<T>::addVertex(const T &in) {
542 if (findVertex(in) != nullptr)
543 return false;
544 vertexSet.push_back(new Vertex(in));
545 return true;
546}

548/*
549 * Removes a vertex with a given content (in) from a graph (this), and
550 * all outgoing and incoming edges.
551 * Returns true if successful, and false if such vertex does not exist.
552 */
553template<class T>
554bool Graph<T>::removeVertex(const T &in) {
555 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
556 if ((*it)->getInfo() == in) {
557 auto v = *it;
558 v->removeOutgoingEdges();
559 for (auto u : vertexSet) {
560 u->removeEdge(v->getInfo());
561 }
562 vertexSet.erase(it);
563 delete v;
564 return true;
565 }
566 }
567 return false;
568}

570/*
571 * Adds an edge to a graph (this), given the contents of the source and
572 * destination vertices and the edge weight (w).
573 * Returns true if successful, and false if the source or destination vertex does not exist.
574 */
575template<class T>
576bool Graph<T>::addEdge(const T &sourc, const T &dest, double c, int domain) {
577 auto v1 = findVertex(sourc);
578 auto v2 = findVertex(dest);
579 if (v1 == nullptr || v2 == nullptr)
580 return false;
581 v1->addEdge(v2, c, domain);
582 return true;
583}

585/*
586 * Removes an edge from a graph (this).
587 * The edge is identified by the source (sourc) and destination (dest) contents.
588 * Returns true if successful, and false if such edge does not exist.
589 */
590template<class T>
591bool Graph<T>::removeEdge(const T &sourc, const T &dest) {
592 Vertex<T>* srcVertex = findVertex(sourc);
593 if (srcVertex == nullptr) {
594 return false;
595 }
596 return srcVertex->removeEdge(dest);
597}

599template<class T>
600Graph<T>::Graph(Data &data) {
601
602 std::vector<Submission> &submissions = data.submissions;
603 std::vector<Reviewer> &reviewers = data.reviewers;
604 const Parameters &p = data.parameters;
605 const Control &c = data.control;
606
607 nodeInfo source = {SOURCE, 0};
608 nodeInfo sink = {SINK, -1};
609
610 /* CREATE THE NODES */
611 // Create the source node ID => 0
612 this->addVertex(source);
613 // Create the sink node ID => -1
614 this->addVertex(sink);
615
616 // Create submission nodes
617 for (const Submission &s : submissions) {
618 this->addVertex({SUBMISSION, s.getId()});
619 }
620 // Create review nodes R
621 for (const Reviewer &r : reviewers) {
622 this->addVertex({REVIEWER, r.getId()});
623 }
624 /* CREATE THE EDGES*/
625 // Create edge from the source node to submission with capacity MinReviewsPerSubmission
626 for (const Submission& s : submissions) {
627 // When checking the outputs the domain of this edge is always the primary Field
628 this->addEdge(source, {SUBMISSION, s.getId()}, p.MinReviewsPerSubmission, s.getPrimaryField());
629 }
630
631 // Create edges between submissions and reviewers
632 for (const Submission &s : submissions) {
633 for (const Reviewer &r : reviewers) {
634 nodeInfo submission = {SUBMISSION, s.getId()};
635 nodeInfo reviewer = {REVIEWER, r.getId()};
636
637 switch (c.GenerateAssignments) {
638 case 0:
639 case 1:
640 // Create only primary edges
641 if (s.getPrimaryField() == r.getPrimaryField()) {
642 this->addEdge(submission, reviewer, 1, r.getPrimaryField());
643 }
644 break;
645
646 case 2:
647 // primary + secondary submissions, primary reviewers only
648 if (s.getPrimaryField() == r.getPrimaryField())
649 this->addEdge(submission, reviewer, 1, r.getSecondaryField());
650
651 // secondary only if both fields exist
652 if (s.getSecondaryField() != 0 && r.getPrimaryField() == s.getSecondaryField())
653 this->addEdge(submission, reviewer, 1, r.getPrimaryField());
654 break;
655
656 case 3:
657 // general case: all primary and secondary
658
659 // primary only if both fields exist
660 if (s.getPrimaryField() == r.getPrimaryField())
661 this->addEdge(submission, reviewer, 1, r.getPrimaryField());
662
663 // primary with secondary only if both fields exist
664 if (r.getSecondaryField() != 0 && s.getPrimaryField() == r.getSecondaryField())
665 this->addEdge(submission, reviewer, 1, r.getSecondaryField());
666
667 // secondary with primary only if both fields exist
668 if (s.getSecondaryField() != 0 && s.getSecondaryField() == s.getPrimaryField())
669 this->addEdge(submission, reviewer, 1, r.getPrimaryField());
670
671 // secondary with secondary only if both fields exist
672 if (s.getSecondaryField() != 0 && r.getSecondaryField() != 0 && s.getSecondaryField() == r.getSecondaryField())
673 this->addEdge(submission, reviewer, 1, r.getSecondaryField());
674 break;
675 default: ;
676 }
677 }
678 }
679
680 // Create edges to sink with capacity equal to the MaxReviewsPerReviewer
681 for (const Reviewer& r : reviewers) {
682 this->addEdge({REVIEWER, r.getId()}, sink, p.MaxReviewsPerReviewer);
683 }
684}
