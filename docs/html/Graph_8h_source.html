413template<class T>
414bool Vertex<T>::removeEdge(const T &id) {
415 bool removedEdge = false;
416 auto it = adj.begin();
417 while (it != adj.end()) {
418 Edge<T> *edge = *it;
419 Vertex *dest = edge->getDest();
420 if (dest->getInfo() == id) {
421 it = adj.erase(it);
422 deleteEdge(edge);
423 removedEdge = true; // not breaking allows for multiple edges to connect the same pair of vertices (multigraph)
424 }
425 else {
426 ++it;
427 }
428 }
429 return removedEdge;
430}

432template<class T>
433void Vertex<T>::removeOutgoingEdges() {
434 auto it = adj.begin();
435 while (it != adj.end()) {
436 Edge<T> *edge = *it;
437 it = adj.erase(it);
438 deleteEdge(edge);
439 }
440}

482template<class T>
483void Vertex<T>::deleteEdge(Edge<T> *edge) {
484 Vertex<T> *dest = edge->getDest();
485 // Remove only the matching edge pointer from the incoming list of the destination node
486 auto it = dest->incoming.begin();
487 while (it != dest->incoming.end()) {
488 if (*it == edge) {
489 it = dest->incoming.erase(it);
490 }
491 else {
492 ++it;
493 }
494 }
495 delete edge;
496}

575template<class T>
576bool Graph<T>::removeVertex(const T &in) {
577 for (auto it = vertexSet.begin(); it != vertexSet.end(); ++it) {
578 if ((*it)->getInfo() == in) {
579 auto v = *it;
580 v->removeOutgoingEdges();
581 for (auto u : vertexSet) {
582 u->removeEdge(v->getInfo());
583 }
584 vertexSet.erase(it);
585 delete v;
586 return true;
587 }
588 }
589 return false;
590}

615void Graph<T>::createNodes(nodeInfo source, nodeInfo sink, const std::vector<Submission> &submissions, const std::vector<Reviewer> &reviewers) {
616 // Create the source node ID => 0
617 this->addVertex(source);
618 // Create the sink node ID => -1
619 this->addVertex(sink);
620
621 // Create submission nodes
622 for (const Submission &s : submissions) {
623 this->addVertex({SUBMISSION, s.getId()});
624 }
625 // Create review nodes R
626 for (const Reviewer &r : reviewers) {
627 this->addVertex({REVIEWER, r.getId()});
628 }
629}

631template<class T>
632void Graph<T>::createEdges(nodeInfo source, nodeInfo sink, const std::vector<Submission> &submissions, const std::vector<Reviewer> &reviewers, const Parameters &parameters, int generateAssignments) {
633 // Create edge from the source node to submission with capacity MinReviewsPerSubmission
634 for (const Submission& s : submissions) {
635 // The domain of source edges is always the primary field of the submission
636 this->addEdge(source, {SUBMISSION, s.getId()}, parameters.MinReviewsPerSubmission, s.getPrimaryField());
637 }
638
639 // Create flags to help in the creation of the edges, regarding what types of edges to consider
640 // Names of the variables are of the type <submissionExpertise><reviewerExpertise>
641 const bool primaryPrimary = parameters.PrimarySubmissionDomain && parameters.PrimaryReviewerExpertise;
642 const bool primarySecondary = parameters.PrimarySubmissionDomain && parameters.SecondaryReviewerExpertise;
643 const bool secondaryPrimary = parameters.SecondarySubmissionDomain && parameters.PrimaryReviewerExpertise;
644 const bool secondarySecondary = parameters.SecondarySubmissionDomain && parameters.SecondaryReviewerExpertise;
645
646 // Create edges between submissions and reviewers
647 for (const Submission &s : submissions) {
648 for (const Reviewer &r : reviewers) {
649 nodeInfo submission = {SUBMISSION, s.getId()};
650 nodeInfo reviewer = {REVIEWER, r.getId()};
651
652 // helper lambda function that reduces code repetition
653 // the variable 'field' will be the domain of the connection submission-reviewer established
654 auto add = [&](int field) {
655 this->addEdge(submission, reviewer, 1, field);
656 };
657
658 switch (generateAssignments) { // in addition to this value the flags above will be used to determine what edges are created
659 case 0:
660 case 1:
661 // Create only primary fields
662 if (primaryPrimary &&
663 s.getPrimaryField() == r.getPrimaryField()) {
664 add(r.getPrimaryField());
665 }
666 break;
667
668 case 2:
669 // primary + secondary submissions, primary reviewers only
670 if (primaryPrimary &&
671 s.getPrimaryField() == r.getPrimaryField())
672 add(r.getPrimaryField());
673
674 // secondary only if both fields exist
675 if (secondaryPrimary &&
676 s.getSecondaryField() != NO_FIELD &&
677 s.getSecondaryField() == r.getPrimaryField())
678 add(r.getPrimaryField());
679 break;
680
681 case 3:
682 // general case: all primary and secondary
683
684 // primary only if both fields exist
685 if (primaryPrimary &&
686 s.getPrimaryField() == r.getPrimaryField())
687 add(r.getPrimaryField());
688
689 // primary with secondary only if both fields exist
690 if (primarySecondary &&
691 r.getSecondaryField() != NO_FIELD &&
692 s.getPrimaryField() == r.getSecondaryField())
693 add(r.getSecondaryField());
694
695 // secondary with primary only if both fields exist
696 if (secondaryPrimary &&
697 s.getSecondaryField() != NO_FIELD &&
698 s.getSecondaryField() == r.getPrimaryField())
699 add(r.getPrimaryField());
700
701 // secondary with secondary only if both fields exist
702 if (secondarySecondary &&
703 s.getSecondaryField() != NO_FIELD &&
704 r.getSecondaryField() != NO_FIELD &&
705 s.getSecondaryField() == r.getSecondaryField())
706 add(r.getSecondaryField());
707 break;
708 default: ;
709 }
710 }
711 }
712
713 // Create edges to sink with capacity equal to the MaxReviewsPerReviewer
714 for (const Reviewer& r : reviewers) {
715 this->addEdge({REVIEWER, r.getId()}, sink, parameters.MaxReviewsPerReviewer);
716 }
717}

719template<class T>
720void Graph<T>::outputGraph(const int numSubmissions, const int numReviewers) const {
721 std::cout << "\nGraph created with " << numSubmissions << " submission nodes, " << numReviewers
722 << " reviewer nodes, and " << this->getNumVertex() << " vertices in total." << std::endl;
723
724 std::cout << "\n=============== Edges ===============\n" << std::endl;

730 }
731 }
